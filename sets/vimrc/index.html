.vimrc

May 15th, 2012

我使用Windows系统，因此_vimrc文件放在C:\Users\Trowa\下。

" Environment => Windows 7 with GUI
" 当使用命令行或者在linux下时本文件可能不适用!
"
set nocompatible

" 编码设定
" vim内部工作在utf-8上
set encoding=utf-8
" 新建文件默认为utf-8无bom
set fileencoding=ucs-bom
" 打开文件时的编码猜测顺序
set fileencodings=ucs-bom,utf-8,cp936,gb18030,big5,euc-jp,euc-kr,latin1 

filetype plugin indent on
syntax on

set nu
set tabstop=4

" 显示当前光标的行列信息
set ruler

" 删掉Gui的工具栏
set guioptions-=T

" 设置窗口大小
set lines=40
set columns=120

" 鼠标遵循windows规则
behave mswin

" 设置界面为英文避免菜单扯不下来的问题
set langmenu=en_US
let $LANG='en_US'
source $VIMRUNTIME/delmenu.vim
source $VIMRUNTIME/menu.vim

colorscheme molokai

" 字体
" guifont是英文字体 gfw是双字节(东亚汉字字体)
set guifont=DejaVu_Sans_Mono:h10:cANSI
set guifontwide=Yahei_Mono:h10

" 快捷键
map <C-TAB> :tabNext<CR>
map <C-T> :tabnew<CR>
map <C-W> :tabc<CR>

配色方案molokai可以在此处找到。
